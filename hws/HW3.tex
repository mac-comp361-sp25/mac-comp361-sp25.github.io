\section*{Questions}
\begin{questions}
    \question \textit{(Based on Sipser 1.44)} We will show, in a number of steps, that for all $k \geq 1$, there exists a language $A_k \subseteq \{0,1\}^*$ such that a DFA with $k$ states can accept it, but a DFA with $k-1$ states cannot.

    \begin{parts}
        \part *Consider the language $A_k = \{w \in \Sigma^* \mid \lvert w \rvert \geq k-1\}$, the set of strings using our alphabet of length $k-1$ or greater. Show that for any $k \geq 1$, you can construct a DFA (by providing a 5-tuple) that will recognize $A_k$ that contains exactly $k$ states.

        \part *Now prove that no DFA with $k-1$ states can recognize $A_k$. 

        \textit{**HINT**: Consider the string $0^{k-1} \in A_k$. Suppose a DFA $M$ with $k-1$ states existed that recognized $A_k$, and thus accepted $0^{k-1}$. Inspect the computation that demonstrates the acceptance of $0^{k-1}$ and arrive at a contradiction.}
    \end{parts}

    \question \textit{Sipser 1.54/1.55} Let $\Sigma = \{a, b\}$. Consider the language $C_k = L(R)$ defined by the regular expression $R = \Sigma^*a\Sigma^{k-1}$. That is, the language of strings with a the symbol $a$ exactly $k$ places from the end.

    \begin{parts}
        \part Provide a 5-tuple defining an NFA with $k+1$ states that recognizes $C_k$
        \part Prove that for each $k \geq 1$, no DFA with fewer than $2^k$ states can recognize $C_k$.

        \textit{**HINT**: Construct an argument similar in spirit to your solution to 1b}
    \end{parts}
\end{questions}
